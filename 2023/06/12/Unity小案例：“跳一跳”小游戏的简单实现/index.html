Unity小案例：“跳一跳”小游戏的简单实现

2023/6/12 Unity

从这篇开始，我要把一些做过的unity小案例搬到这上面来^_^

一个模仿微信跳一跳小游戏的案例，其实是照着b站的视频做的，但是原视频是在找不到了，这里只能不贴了其实是学姐偷懒找我帮做的结课作业

using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UIElements;

public class PlayerControl : MonoBehaviour
{
    Rigidbody rb;
    float timer;
    public float maxChargeTime;
    public float jumpPower;

    public bool doJump;
    public bool isGround;

    void Start()
    {
        rb = GetComponent<Rigidbody>();
    }

    void Update()
    {
        if (Input.GetKey(KeyCode.Mouse0) && timer < maxChargeTime)
        {
            timer += Time.deltaTime;
            transform.localScale = new Vector3(1, 0.5f + (maxChargeTime - timer) / (2 * maxChargeTime), 1);
        }

        if (Input.GetKeyUp(KeyCode.Mouse0))
        {
            doJump = true;
            transform.localScale = Vector3.one;
        }

        if(transform.position.y<-0.5f)
        {
            GameManage.S.gameOver = true;
        }
    }

    void FixedUpdate()
    {
        if (doJump && isGround)
        {
            Jump();
            doJump = false;
            isGround = false;
            timer = 0;
        }
    }

    void Jump()
    {
        Vector3 dir;
        if (GameManage.S.playerIsFacingYAxis)
        {
            dir = transform.forward;
        }
        else
        {
            dir = -transform.right;
        }
        dir.y = 1;
        rb.velocity = dir * timer * jumpPower;
    }

    void OnCollisionEnter(Collision collision)
    {
        if (collision.gameObject.CompareTag("Ground"))
        {
            if (collision.contacts[0].normal == Vector3.up)
            {
                isGround = true;
                GameManage.S.HitGround(transform.position);
            }
        }
    }
}
